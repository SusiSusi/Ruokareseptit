4   import ruokareseptit.domain.Ainesosa;

6   /**
7    * Luokka vertailee sanoja toisiinsa ja muokkaa ainesosa-listan tiedoston
8    * kirjoittamiseen sopivaksi, tavuttaa reseptin ohjeen, tulostaa tähtiä 
9    * sekä tulostaa HTML-rivien vaihtoja. "Kielenhoito"-luokka
10   *
11   * @author susisusi
12   */
13  public class StringUtils {
14  
15      /**
16       * Metodi vertaa annettuja sanoja toisiinsa tarkoittavatko ne samaa.
17       * Metodissa poistetaan sanojen turhat välilyönnit sekä muutetaan sanojen
18       * kirjaimet isoiksi kirjaimiksi.
19       *
20       * @param sana tietokannassa olevan sanan virallinen esitysmuoto
21       * @param haettava Käyttäjän antama syöte
22       * @return totuusarvon, onko sana sama vai ei
23       */
24      public boolean sisaltaa(String sana, String haettava) {
25          if (sana.isEmpty() || haettava.isEmpty()) {
26              return false;
27          }
28          String trimSana = sana.trim();
29          String trimHaettava = haettava.trim(); // poistetaan turhat välilyönnit
30  
31          trimSana = trimSana.toUpperCase();
32          trimHaettava = trimHaettava.toUpperCase(); // muutetaan kaikki kirjaimet isoiksi
33          return trimSana.equals(trimHaettava);
34      }
35  
36      /**
37       * Metodi muuttaa annetun ainesosa-listan tiedostoon sopivaksi. Tiedostoon
38       * kirjoittaminen edellyttää, että ainesosat ovat kirjoitettu muotoon
39       * maara:aines, esim. 1 kpl:sipuli
40       *
41       * @param ainesosat reseptin ainesosat
42       * @return ainesosat String-muodossa
43       */
44      public String muutaAinesosatTiedostoonSopiviksi(List<Ainesosa> ainesosat) {
45          String muutettu = "";
46          for (Ainesosa ainesosa : ainesosat) {
47              String maara = ainesosa.getMaara();
48              String aines = ainesosa.getNimi();
49              muutettu = muutettu + maara + ":" + aines + "\n";
50          }
51          return muutettu;
52      }
53  
54      /**
55       * Palauttaa rivitetyn version reseptin ohjeesta. Kun rivin pituus on
56       * 80-merkkiä, tulostaa metodi riville 80-merkissä olevan viimeisen sanan ja
57       * aloittaa uuden rivin eli metodi ei katkaise sanoja vaan rivi katkeaa kun
58       * seuraava merkki on tyhjä.
59       *
60       * @param ohje reseptin ohje
61       * @return muokatun ohjeen
62       */
63      public String tavutaReseptinOhje(String ohje) {
64          String muokattuOhje = "";
65          if (ohje.equals("Ohjetta ei ole talletettu.")) {
66              return ohje;
67          } else if (ohje.equals("Kirjoita tähän valmistusohje")) {
68              return "Ohjetta ei ole talletettu.";
69          } else {
70              int pituus = ohje.length();
71              int rivinPituus = 80;
72              int apu = 0;
73              String rivi = "";
74              for (int i = 0; i < pituus; i++) {
75                  rivi = rivi + ohje.charAt(i);
76                  apu++;
77                  if (apu >= rivinPituus && ohje.charAt(i) == ' ') {
78                      muokattuOhje = muokattuOhje + rivi + "\n";
79                      rivi = "";
80                      apu = 0;
81                  }
82              }
83              muokattuOhje = muokattuOhje + rivi;
84          }
85          return muokattuOhje;
86      }
87  
88      /**
89       * Metodi tulostaa tähtirivit reseptien erotteluun toisistaan.
90       * @param maara kuinka monta tähteä riviin halutaan
91       * @return palauttaa tähtirivin
92       */
93      public String tulostaTahtia(int maara) {
94          String tahtia = "";
95          for (int i = 0; i < maara; i++) {
96              tahtia = tahtia + "*";
97          }
98          return tahtia;
99      }
100     
101     /**
102      * Metodin avulla voidaan tulostaa HTML-rivivaihtoja. Metodia tarvitaan 
103      * graafisen käyttöliittymän kanssa.
104      * @param maara monta rivivaihtoa halutaan
105      * @return palauttaa rivin rivivaihtoja
106      */
107     public String htmlRiviVaihtoja(int maara) {
108         String riviVaihdot = "";
109         for (int i = 0; i < maara; i++) {
110             riviVaihdot = riviVaihdot + "<br>";
111         }
112         return riviVaihdot;
113     }
